existing.ref_code : refCode, already_registered: true }), { headers: corsHeaders }); } } if (request.method === 'GET' && path === '/leaderboard') { const results = await env.DB.prepare( 'SELECT s.ref_code, SUBSTR(s.email, 1, INSTR(s.email, \'@\') - 1) as name, COUNT(r.id) as referral_count FROM signups s LEFT JOIN referrals r ON s.ref_code = r.ref_code GROUP BY s.ref_code ORDER BY referral_count DESC LIMIT 10' ).all(); return new Response(JSON.stringify({ leaderboard: results.results }), { headers: corsHeaders }); } const count = await env.STREAMSTACK.get('waitlist_count'); return new Response(JSON.stringify({ count: parseInt(count) || 0 }), { headers: corsHeaders }); }, };
